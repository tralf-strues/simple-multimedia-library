129 void Renderer::renderTexture(const Texture& texture,
130 const Rectangle<int32_t>* targetRegion,
131 const Rectangle<int32_t>* sourceRegion)
132 {
133 SDL_Rect targetRect;
134 SDL_Rect sourceRect;
135
136 SDL_Rect* pTargetRect = nullptr;
137 SDL_Rect* pSourceRect = nullptr;
138
139 if (targetRegion != nullptr)
140 {
141 targetRect = toNativeRectangle(*targetRegion);
142 pTargetRect = &targetRect;
143 }
144
145 if (sourceRegion != nullptr)
146 {
147 sourceRect = toNativeRectangle(*sourceRegion);
148 pSourceRect = &sourceRect;
149 }
150
151 SDL_RenderCopy(m_NativeRenderer, texture.getNativeTexture(), pSourceRect, pTargetRect);
152 }
