127void Renderer::renderTexture(const Texture& texture,
128 const Rectangle<int32_t>* targetRegion,
129 const Rectangle<int32_t>* sourceRegion)
130{
131 SDL_Rect targetRect;
132 SDL_Rect sourceRect;
133
134 SDL_Rect* pTargetRect = nullptr;
135 SDL_Rect* pSourceRect = nullptr;
136
137 if (targetRegion != nullptr)
138 {
139 targetRect = toNativeRectangle(*targetRegion);
140 pTargetRect = &targetRect;
141 }
142
143 if (sourceRegion != nullptr)
144 {
145 sourceRect = toNativeRectangle(*sourceRegion);
146 pSourceRect = &sourceRect;
147 }
148
149 SDL_RenderCopy(m_NativeRenderer, texture.getNativeTexture(), pSourceRect, pTargetRect);
150}
