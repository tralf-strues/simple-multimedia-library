41 Renderer::Renderer(Window* window) : m_Window(window)
42 {
43 assert(window);
44
45 m_NativeRenderer = SDL_CreateRenderer(m_Window->getNativeWindow(),
46 -1,
47 SDL_RENDERER_ACCELERATED | SDL_RENDERER_TARGETTEXTURE);
48 if (m_NativeRenderer == nullptr)

50 setError(Renderer::CREATE_ERROR);

210 Color* Renderer::readTargetPixels(const Rectangle<int32_t>* targetRegion) const
211 {
212 Rectangle<int32_t> realRegion = (targetRegion != nullptr) ?
213 *targetRegion :
214 Rectangle<int32_t>{{0, 0},
215 static_cast<int32_t>(getTargetWidth()),
216 static_cast<int32_t>(getTargetHeight())};
217
218 if (realRegion.width <= 0 || realRegion.height <= 0)
219 {
220 return nullptr;
221 }
222
223 Color* pixels = new Color[realRegion.width * realRegion.height];
224 readTargetPixelsTo(pixels, targetRegion);
225
226 return pixels;
227 }

229 void Renderer::readTargetPixelsTo(Color* dst, const Rectangle<int32_t>* targetRegion) const
230 {
231 assert(dst);
232
233 SDL_Rect rect;
234 if (targetRegion != nullptr)
235 {
236 rect = {targetRegion->pos.x, targetRegion->pos.y, targetRegion->width, targetRegion->height};
237 }
238
239 SDL_RenderReadPixels(getNativeRenderer(),
240 targetRegion == nullptr ? nullptr : &rect,
241 SDL_PIXELFORMAT_RGBA8888,
242 dst,
243 (targetRegion == nullptr ? getTargetWidth() : rect.w) * sizeof(Color));
244 }

275 void Renderer::renderTexture(const Texture& texture,
276 const Rectangle<int32_t>* targetRegion,
277 const Rectangle<int32_t>* sourceRegion)
278 {
279 SDL_Rect targetRect;
280 SDL_Rect sourceRect;
281
282 SDL_Rect* pTargetRect = nullptr;
283 SDL_Rect* pSourceRect = nullptr;
284
285 if (targetRegion != nullptr)
286 {
287 targetRect = toNativeRectangle(*targetRegion);
288 pTargetRect = &targetRect;
289 }
290
291 if (sourceRegion != nullptr)
292 {
293 sourceRect = toNativeRectangle(*sourceRegion);
294 pSourceRect = &sourceRect;
295 }
296
297 SDL_RenderCopy(m_NativeRenderer, texture.getNativeTexture(), pSourceRect, pTargetRect);
298 }
