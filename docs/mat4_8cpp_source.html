74Mat4<float> lookAt(const Vec3<float>& from, const Vec3<float>& direction)
75{
76 static const Vec3<float> tmp = {0, 1, 0};
77
78 Vec3<float> forward = normalize(direction);
79 Vec3<float> right = crossProduct(tmp, forward);
80 Vec3<float> up = crossProduct(forward, right);
81
82 return {{right.x, right.y, right.z, -dotProduct(right, from),
83 up.x, up.y, up.z, -dotProduct(up, from),
84 forward.x, forward.y, forward.z, -dotProduct(forward, from),
85 0, 0, 0, 1 }};
