16class EventDispatcher

19 //--------------------------------------------------------------------------
20 //! @brief Dispatches the event.
21 //!
22 //! Proccesses the event and/or its further path down the
23 //! @ref EventDispatchChain if necessary. If the event is not consumed
24 //! during the capturing phase, it is supposed to be dispatcher to the rest
25 //! of the chain.
26 //!
27 //! @param event

29 //! the whole dispatching proccess.
30 //!
31 //! @return The return event or nullptr if the event has been consumed.
32 //--------------------------------------------------------------------------
33 virtual Event* dispatchEvent(Event* event, EventDispatchChain* chain);
34};

36class EventDispatchChain
37{
38public:
39 //--------------------------------------------------------------------------
40 //! @brief Prepends the dispatcher to the chain, i.e. adds the dispatcher to
41 //! the front of the chain (before the chain's current head).
42 //!
43 //! @attention The caller shouldn't assume that this @ref EventDispatchChain
44 //! remains unchanged nor that the returned chain will point to a
45 //! different chain after the call. The call look like this:
46 //! @code chain = chain.prependDispatcher(dispatcher); @endcode
47 //!
48 //! @param dispatcher
49 //--------------------------------------------------------------------------
50 virtual void prependDispatcher(EventDispatcher* dispatcher);
51
52 //--------------------------------------------------------------------------
53 //! @brief Send the event through this chain.
54 //!
55 //! The event is passed from one @ref EventDispatcher to the next in the
56 //! chain until the end of the chain is reached. Each @ref EventDispatcher
57 //! in the chain can change the event path and the event itself. The chain
58 //! is usually formed by following some hierarchy from the root to the
59 //! event target (or vice versa in the internal SGL implementation).
60 //!
61 //! @param event
62 //!
63 //! @return The proccessed event or nullptr if the event has been consumed.
64 //--------------------------------------------------------------------------
65 virtual Event* sendThroughChain(Event* event);
66};
